1. Resume Materi Constructor & Enkapsulasi

1.1 Pengertian Constructor

Constructor adalah metode khusus dalam kelas yang digunakan untuk menginisialisasi objek saat dibuat. Nama constructor sama dengan nama kelas dan tidak memiliki tipe kembalian.

1.2 Jenis-Jenis Constructor

  • Default Constructor: Constructor tanpa parameter, dibuat secara otomatis jika tidak ada constructor lain.
  • Parameterized Constructor: Constructor dengan parameter untuk mengisi nilai atribut langsung.
  • Named Constructor: Constructor dengan nama khusus untuk membuat objek dengan cara berbeda.
  • Constant Constructor: Digunakan untuk membuat objek immutable (nilai tidak bisa diubah) dengan keyword const.
  • Constructor dengan Initializer List: Menginisialisasi atribut sebelum body constructor dijalankan.

1.3 Pengertian Enkapsulasi

Enkapsulasi adalah konsep menyembunyikan data dengan membuat atribut menjadi private (awalan _), lalu mengatur aksesnya melalui getter dan setter.

1.4 Komponen Enkapsulasi

  • Atribut Private: Tidak bisa diakses langsung dari luar kelas.
  • Getter: Metode untuk membaca nilai atribut private.
  • Setter: Metode untuk mengubah nilai atribut private dengan validasi.

class Mahasiswa {
  String? nama;
  int? nim;
  String? jurusan;
  String? kelas;

  // Constructor Default
  Mahasiswa();

  // Named Constructor dariMap
  Mahasiswa.dariMap(Map data) {
    nama = data['nama'];
    nim = data['nim'];
    jurusan = data['jurusan'];
    kelas = data['kelas'];
  }

  // Named Constructor tanpaNama
  Mahasiswa.tanpaNama() {
    nama = "Tanpa Nama";
    nim = 00000000;
    jurusan = "Tidak Diketahui";
    kelas = "Tidak Diketahui";
  }

  void tampilData() {
    print("=== DATA MAHASISWA ===");
    print("Nama    : $nama");
    print("NIM     : $nim");
    print("Jurusan : $jurusan");
    print("Kelas   : $kelas\n");
  }
}

void main() {
  // Objek dari Default Constructor

class User {
  String? _username;
  String? _password;
  String? _email;

  // Constructor
  User(this._username, this._password, this._email);

  // Getter
  String? get username => _username;
  String? get email => _email;

  // Setter dengan Validasi
  set password(String? value) {
    if (value != null && value.length >= 8) {
      _password = value;
      print("✅ Password berhasil diubah!");
    } else {
      print("❌ Password minimal 8 karakter!");
    }
  }

  set username(String? value) {
    if (value != null && value.length >= 5) {
      _username = value;
      print("✅ Username berhasil diubah!");
    } else {
      print("❌ Username minimal 5 karakter!");
    }
  }

  set email(String? value) {
    if (value != null && value.contains('@')) {
      _email = value;
      print("✅ Email berhasil diubah!");
    } else {
      print("❌ Email harus mengandung @!");
    }
  }

  // Method Login
  bool login(String inputUser, String inputPass) {
    return _username == inputUser && _password == inputPass;
  }
}

class Produk {

  // Getter
  String? get nama => _nama;
  double? get harga => _harga;

  // Setter dengan Validasi
  set nama(String? value) {
    if (value != null && value.length >= 3) {
      _nama = value;
    } else {
      print("❌ Nama produk minimal 3 karakter!");
    }
  }

  set harga(double? value) {
    if (value != null && value > 0) {
      _harga = value;
    } else {
      print("❌ Harga harus lebih besar dari 0!");
    }
  }

  void tampilProduk() {
    print("\n=== DATA PRODUK ===");
    print("Nama Produk: $_nama");
    print("Harga Produk: $_harga");
  }
}
